41  def calc(self, j, m2, m1, beta):
42  sgn = 1
43 
44  if m1 + m2 < 0:
45  m1 *= -1
46  m2 *= -1
47  sgn *= (-1)**(m2 - m1)
48 
49  if m1 < m2:
50  m1, m2 = m2, m1
51  sgn *= (-1)**(m2 - m1)
52 
53  bstring = "{:+013d}.pkl".format(int(np.round(beta * 1e9)))
54  if bstring not in self.wignerdict.keys():
55  path = os.path.join(self.cachedir, bstring)

59  else:

63  mstring = "{}_{}_{}".format(j, m1, m2)

65  self.wignerdict[bstring][mstring] = float(
66  np.real(self._eval_wignerd(j, m2, m1, beta)))

103  def _m_values(self, j):
104  size = 2 * j + 1
105  if not size.is_integer() or not size > 0:
106  raise ValueError(
107  'Only integer or half-integer values allowed for j, got: : %r' % j
108  )
109  return size, [j - i for i in range(int(2 * j + 1))]

111  def _eval_wignerd(self, j, m, mp, beta):
112  from numpy import pi, sin, cos, sqrt
113  from scipy.misc import factorial
114  from scipy.special import binom as binomial
115 
116  r = 0
117  if beta == pi / 2:
118  # Varshalovich Equation (5), Section 4.16, page 113, setting
119  # alpha=gamma=0.
120  for k in range(int(2 * j) + 1):
121  if k > j + mp or k > j - m or k < mp - m:
122  continue
123  r += (-1)**k * binomial(j + mp, k) * \
124  binomial(j - mp, k + m - mp)
125  r *= (-1)**(m - mp) / 2**j * \
126  sqrt(factorial(j + m) * factorial(j - m) /
127  (factorial(j + mp) * factorial(j - mp)))
128  else:
129  # Varshalovich Equation(5), Section 4.7.2, page 87, where we set
130  # beta1=beta2=pi/2, and we get alpha=gamma=pi/2 and beta=phi+pi,
131  # then we use the Eq. (1), Section 4.4. page 79, to simplify:
132  # d(j, m, mp, beta+pi) = (-1)**(j-mp) * d(j, m, -mp, beta)
133  # This happens to be almost the same as in Eq.(10), Section 4.16,
134  # except that we need to substitute -mp for mp.
135  size, mvals = self._m_values(j)
136  for mpp in mvals:
137  r += self._eval_wignerd(j, m, mpp, pi / 2) * \
138  (cos(-mpp * beta) + 1j * sin(-mpp * beta)) * \
139  self._eval_wignerd(j, mpp, -mp, pi / 2)
140  # Empirical normalization factor so results match Varshalovich
141  # Tables 4.3-4.12
142  # Note that this exact normalization does not follow from the
143  # above equations
144  r = r * 1j**(2 * j - m - mp) * (-1)**(2 * m)
145 
146  return r

152 def csr_vappend(a, b):
153  """ Takes in 2 csr_matrices and appends the second one to the bottom of the first one.
154  Much faster than scipy.sparse.vstack but assumes the type to be csr and overwrites
155  the first matrix instead of copying it. The data, indices, and indptr still get copied."""
156 
157  a.data = np.hstack((a.data, b.data))
158  a.indices = np.hstack((a.indices, b.indices))
159  a.indptr = np.hstack((a.indptr, (b.indptr + a.nnz)[1:]))
160  a._shape = (a.shape[0] + b.shape[0], b.shape[1])
