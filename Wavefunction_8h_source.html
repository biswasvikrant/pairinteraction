28  /** \brief Calculate the model potentials.

30  * This function calculates the model potentials and returns \f$
31  * V_{\mathrm{mod}}(r) = V_{\mathrm{C}}(r) + V_{\mathrm{P}}(r) +
32  * V_{\mathrm{s.o.}}(r) \f$
33  *
34  * \f[
35  * \begin{aligned}
36  * V_{\mathrm{C}}(r) &= - \frac{e^2}{4\pi\varepsilon_0} \frac{1 + (Z-1)\mathrm{e}^{-\alpha_1 r} - r(\alpha_3+\alpha_4 r)\mathrm{e}^{-\alpha_2 r}}{r} \\
37  * V_{\mathrm{P}}(r) &= -\frac{e^2}{(4\pi\varepsilon_0)^2} \frac{\alpha_d}{2 r^4} \left[ 1 - \mathrm{e}^{-(r/r_c)^6} \right] \\

39  * \end{aligned}
40  * \f]
41  *
42  * \param[in] qd Quantum defect data (parameters for the potentials)
43  * \param[in] x Radial position
44  * \returns Model potentials evaluted at position \p x with parameters \p qd
45  */

64 /** \brief %Numerov's method
65  *
66  * This class implements %Numerov's method using a couple of helper functions.
67  * %Numerov's method solves a differential equation of the form
68  * \f[
69  * y''(x) + g(x)y(x) = 0
70  * \f]
71  * The equation is solved by iteratively computing
72  * \f[
73  * y_{n+1} = \frac{\left( 2 + \frac{5 h^2}{6} g_n \right) y_n - \left( 1 + \frac{h^2}{12} g_{n-1} \right) y_{n-1}}{\left( 1 + \frac{h^2}{12} g_n \right)}
74  * \f]
75  *
76  * with \f$ y_n = y(x_n) \f$ and \f$ g_n = g(x_n) \f$. Clearly this equation
77  * needs two initial conditions \f$ y_0 \f$ and \f$ y_1 \f$. We integrate the
78  * equation from outside to inside. Since the wavefunction hs to decay to zero
79  * at infinity we choose \f$ y_0 = 0 \f$. Now depending on the number of knots
80  * of wavefunction we decide whether to set \f$ y_1 = \pm\varepsilon \f$.
81  */

84  std::vector<real_t> x;
85 public:
86  /** \brief Integration step size */
87  real_t const dx;

90  *
91  * The constructor initializes the \p qd member and sets the initial
92  * condition for the integrator. It determines the outer and inner
93  * integration bounds by semiclassical and empirical arguments.
94  *
95  * \param[in] qd Quantum defect data (parameters for the potentials)
96  */
97  Numerov(QuantumDefect const& qd);

232 /** \brief Find and return index
233  *
234  * Find a value in a random access container and return its index in the
235  * container to the caller. This is intended to work with std::vector.
236  *

238  * \param[in] d value to find

248 /** \brief Compute radial matrix elements
249  *
250  * The radial matrix element can be calculated from the integral
251  * \f[
252  * \langle nlj| \hat{p}^{\mathrm{rad}}_\kappa |n'l'j' \rangle =
253  * \int \Psi^{\text{rad}}_{nlj}(r) \Psi^{\text{rad}}_{n'l'j'}(r) r^{2+\kappa} \; dr
254  * \f]
255  * The part \f$ r^{2+\kappa} \f$ varies with rescaling the domain and is this
256  * given by the power_kernel function.
257  *
258  * \tparam T Method for calculating the wavefunction
259  * \param[in] qd1 Quantum defect data for first atom
260  * \param[in] power Exponent kappa
261  * \param[in] qd2 Quantum defect data for second atom
262  * \returns Radial matrix element
263  */
264 template < typename T >
265 real_t IntegrateRadialElement( QuantumDefect const& qd1, int power, QuantumDefect const& qd2)
266 {
267  T N1(qd1);
268  T N2(qd2);
269 
270  auto const& x1 = N1.axis();
271  auto const& y1 = N1.integrate();
272  auto const& x2 = N2.axis();
273  auto const& y2 = N2.integrate();
274  auto const dx = N1.dx;
275 
276  auto const xmin = x1.front() >= x2.front() ? x1.front() : x2.front();
277  auto const xmax = x1.back() <= x2.back() ? x1.back() : x2.back();
278 
279  real_t mu = 0;
280  // If there is an overlap, calculate the matrix element
281  if (xmin <= xmax) {
282  int start1 = findidx(x1, xmin);
283  int end1 = findidx(x1, xmax);
284  int start2 = findidx(x2, xmin);
285  int end2 = findidx(x2, xmax);
286 
287  int i1, i2;
288  for (i1 = start1, i2 = start2; i1 < end1 && i2 < end2; ++i1, ++i2)
289  {
290  mu += y1[i1]*y2[i2] * std::pow(x1[i1], T::power_kernel(power)) * dx;
291  }
292  mu = std::abs(2*mu);
293  }
294 
295  return mu;
296 }
